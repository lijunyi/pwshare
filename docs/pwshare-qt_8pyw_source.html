187  for qm in lang_list:
188  self.ui.cmb_lang.addItem(qm)

232  @qt.pyqtSlot()
233  def start_wifi(self):
234  # create wifi with ssid and key
235  msg = self.get_time() + self.tr("Starting Wifi......\n")
236  msg_error = self.get_time() + self.tr("Error: check Admin!\n")
237  ret = self.__ws.create_wifi(s = self.ui.txt_ssid.text(), k = self.ui.txt_key.text())
238  if 0 == ret[0]:
239  # start the created wifi
240  msg += self.get_time() + ret[1]
241  ret = self.__ws.start_wifi()
242  if 0 == ret[0]:
243  # show message
244  msg += self.get_time() + ret[1]
245  msg += self.get_time() + self.tr("Wifi started\n")
246  self.ui.txt_status.setPlainText(msg)
247  self.ui.btn_start.released.disconnect(self.start_wifi)
248  self.ui.btn_start.released.connect(self.close_wifi)
249  self.ui.set_btn_start_text(True)
250  return True
251  else:
252  msg_error += self.get_time() + ret[1]
253  else:
254  msg_error += self.get_time() + ret[1]
255  # failed to create or start wifi
256  self.ui.txt_status.setPlainText(msg_error)
257  return False

273  msg_error += self.get_time() + ret[1]
274  # failed to close wifi
