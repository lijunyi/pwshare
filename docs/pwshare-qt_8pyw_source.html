248  @qt.pyqtSlot()
249  def start_wifi(self):
250  # create wifi with ssid and key
251  msg = self.get_time() + self.tr("Starting Wifi......\n")
252  msg_error = self.get_time() + self.tr("Error: check Admin!\n")
253  ret = self.__ws.create_wifi(s = self.ui.txt_ssid.text(), k = self.ui.txt_key.text())
254  if 0 == ret[0]:
255  # start the created wifi
256  msg += self.get_time() + ret[1]
257  ret = self.__ws.start_wifi()
258  if 0 == ret[0]:
259  # show message
260  msg += self.get_time() + ret[1]
261  msg += self.get_time() + self.tr("Wifi started\n")
262  self.ui.txt_status.setPlainText(msg)
263  self.ui.btn_start.released.disconnect(self.start_wifi)
264  self.ui.btn_start.released.connect(self.close_wifi)
265  self.ui.set_btn_start_text(True)
266  return True
267  else:
268  msg_error += self.get_time() + ret[1]
269  else:
270  msg_error += self.get_time() + ret[1]
271  # failed to create or start wifi
272  self.ui.txt_status.setPlainText(msg_error)
273  return False
